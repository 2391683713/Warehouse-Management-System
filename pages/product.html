// 加载商品列表 async function loadProducts(params) { const result = await API.products.getList(params); if (result.success) { const products = result.data; const tbody = document.querySelector('#productTable tbody'); if (products.length === 0) { tbody.innerHTML = '暂无数据'; return; } tbody.innerHTML = products.map(p => ` ${p.id} ${p.name} ${p.category} ${p.unit} ${p.stock} ¥${p.price} ${p.supplier || '-'} `).join(''); } }
